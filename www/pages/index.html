Declarative viewmodels and bindings

Declarative viewmodels makes it trivial to connect JavaScript to your HTML. Use data-viewmodel="path/to/viewmodel" to apply a viewmodel to any HTML element. By using KnockoutJS properties in the viewmodel can be bound to child elements

  • page.html
  • FirstVM.js
<div data-viewmodel="FirstVM">
  <label>First name</label><br>
  <input data-bind="value: firstName, valueUpdate:'afterkeydown'"><br>
  <label>Last name</label><br>
  <input data-bind="value: lastName, valueUpdate:'afterkeydown'"><br>
  <span data-bind="text: fullName"></span>
</div>
        
define(['knockout'], function(ko){
  return function FirstVM(){
    this.firstName = ko.observable("");
    this.lastName = ko.observable("");
    this.fullName = ko.computed(function(){
      return this.firstName() + " " + this.lastName();
    }, this);
  };
});
